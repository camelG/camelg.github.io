  • laravel url host 取得

    • リファレンス URL
      • ホスト
        • URL ヘルパー

        # laravel url host

        # リファレンス URL

        // リファレンスURL
$ref = request()->server('HTTP_REFERER');

// ドメイン
$domain = @explode('/', str_replace(['http://', 'https://'], '', $ref))[0];

        # ホスト

        $host = request()->getHttpHost(); // With port if there is. Eg: mydomain.com:81

$host = request()->getHost(); // Only hostname Eg: mydomain.com

$host = request()->getSchemeAndHttpHost(); // returns https://dev.site.com

$host = request()->getHttpHost(); // returns dev.site.com

        # URL ヘルパー

        url ヘルパにパスを指定しないと、Illuminate\Routing\UrlGenerator インスタンスが返され、現在の URL に関する情報へアクセスできます。

        // クエリ文字列を除いた現在のURL
echo url()->current();

// クエリ文字列を含んだ現在のURL
echo url()->full();

// 直前のリクエストの完全なURL
echo url()->previous();

        こうしたメソッドには、URL ファサードを使用してもアクセスできます。

        use Illuminate\Support\Facades\URL;

echo URL::current();
